# Считываем количество элементов в массиве
N = int(input())

# Считываем элементы массива
arr = list(map(int, input().split()))

# Инициализируем счетчик количества элементов, больших предыдущего
count = 0

# Проходим по массиву начиная с индекса 1
for i in range(1, N):
    # Если текущий элемент больше предыдущего, увеличиваем счетчик
    if arr[i] > arr[i - 1]:
        count += 1

# Выводим количество элементов, больших предыдущего
print(count)
