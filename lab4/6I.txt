# Считываем количество элементов в массиве
N = int(input())

# Считываем элементы массива
arr = list(map(int, input().split()))

# Сохраняем последний элемент во временной переменной
temp = arr[N - 1]

# Сдвигаем элементы массива вправо
for i in range(N - 1, 0, -1):
    arr[i] = arr[i - 1]

# Первый элемент становится равным временной переменной
arr[0] = temp

# Выводим массив после сдвига
for element in arr:
    print(element, end=" ")
