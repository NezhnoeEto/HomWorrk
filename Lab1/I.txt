# Считываем трехзначное число
number = int(input())

# Извлекаем отдельные цифры
# Сначала находим сотни
hundreds = number // 100

# Затем находим десятки
tens = (number // 10) % 10

# И, наконец, находим единицы
ones = number % 10

# Вычисляем сумму цифр
sum_of_digits = hundreds + tens + ones

# Выводим результат
print(sum_of_digits)
