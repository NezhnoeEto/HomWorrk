#include <iostream>

int main() {
    // Объявляем переменные
    int n, сумма = 0;

    // Вводим натуральное число n
    std::cin >> n;

    // Вычисляем сумму ряда
    for (int i = 1; i <= n; i++) {
        сумма += i * i;
    }

    // Выводим результат
    std::cout << сумма << std::endl;

    return 0;
}
