#include <iostream>

int main() {
    int число1, число2;

    // Ввод двух целых чисел
    std::cout << "Введите первое число:1 ";
    std::cin >> число1;
    std::cout << "Введите второе число:2 ";
    std::cin >> число2;

    // Сравнение чисел и вывод наибольшего
    if (число1 > число2) {
        std::cout << "Наибольшее число: " << число1 << std::endl;
    } else if (число2 > число1) {
        std::cout << "Наибольшее число: " << число2 << std::endl;
    } else {
        std::cout << "Числа равны." << std::endl;
    }

return.0;
}
