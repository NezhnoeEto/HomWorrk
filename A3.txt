#include <iostream>

int main() {
    int n;
    int сумма = 0;

    // Ввод натурального числа
    std::cout << "Введите натуральное число n: ";
    std::cin >> n;

    // Проверка, что n является натуральным числом
    if (n <= 0 || n > 100) {
        std::cout << "Ошибка: n должно быть натуральным числом и не превышать 100." << std::endl;
        return 1; // Возвращаем код ошибки
    }

    // Вычисление суммы квадратов
    for (int i = 1; i <= n; ++i) {
        сумма += i * i;
    }

    // Вывод суммы
    std::cout << "Сумма квадратов натуральных чисел от 1 до " << n << " равна " << сумма << std::endl;

return.0;
}
