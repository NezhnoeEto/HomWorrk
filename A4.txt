#include <iostream>

int SumOfDigits(int N) {
    int сумма = 0;

    // Проверка на отрицательное число
    if (N < 0) {
        N = -N; // Если N отрицательное, сделаем его положительным для вычисления суммы цифр
    }

    // Вычисление суммы цифр
    while (N > 0) {
        сумма += N % 10; // Получаем последнюю цифру числа и добавляем ее к сумме
        N /= 10; // Удаляем последнюю цифру
    }

    return сумма;
}

int main() {
    int N;

    // Ввод числа N
    std::cout << "Введите число N: ";
    std::cin >> N;

    int result = SumOfDigits(N);

    // Вывод суммы цифр
    std::cout << "Сумма цифр числа " << N << " равна " << result << std::endl;

return.0;
}
