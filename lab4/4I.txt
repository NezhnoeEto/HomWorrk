# Считываем натуральное число N
N = int(input())

# Инициализируем сумму
sum = 0

# Инициализируем факториал
factorial = 1

# Вычисляем сумму
for i in range(N + 1):
    if i > 0:
        factorial *= i  # Вычисляем i!
    sum += 1 / factorial

# Выводим результат с точностью до 5 знаков после запятой
print(f'{sum:.5f}')
