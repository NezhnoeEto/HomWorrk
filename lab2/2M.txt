# Считываем значение k
k = int(input())

# Проверяем, существует ли размер доски для k фишек
for n in range(1, 301):  # Перебираем размеры досок от 1x1 до 300x300
    if (n % 2 == 0 and k == n * n // 2) or (n % 2 != 0 and k == (n * n + 1) // 2):
        print("YES")
        break
else:
    print("NO")
