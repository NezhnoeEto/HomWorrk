import math

# Считываем коэффициенты a, b и c
a = float(input())
b = float(input())
c = float(input())

# Вычисляем дискриминант
D = b**2 - 4*a*c

# Проверяем значение дискриминанта
if D > 0:
    # Два корня
    x1 = (-b - math.sqrt(D)) / (2*a)
    x2 = (-b + math.sqrt(D)) / (2*a)
    print(x1, x2)
elif D == 0:
    # Один корень
    x = -b / (2*a)
    print(x)
else:
    # Нет действительных корней
    pass
